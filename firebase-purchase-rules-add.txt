// Purchases Subcollection - Kaufhistorie
match /users/{parent}/purchases/{document} {
  allow create: if request.auth != null;
  allow read: if request.auth != null && request.auth.uid == parent;
  allow write: if request.auth != null && request.auth.uid == parent;
  allow delete: if request.auth != null && request.auth.uid == parent;
}

match /{path=**}/purchases/{document} {
  allow read: if request.auth != null;
}
