// Key principle (CASHBACK_ARCHITECTURE.md §11 #3):
//   ALL writes that affect cashback balance happen via Cloud Functions
//   using Admin SDK. Clients can READ their own data but never WRITE
//   to receipts / payouts / ledger / config. The only client write is
//   the consent block on their own user doc.
//
// Project: markendetektive-895f7
// Pre-req: Phase 1 deploy — these rules can be added the moment the
//          collections are first referenced; they're additive.

// ─── ADD on top-level (outside users) ────────────────────────────────

// Cashback config — public read (so the app knows the tiers + cap),
// admin-only write.
match /cashback_config/{configId} {
  allow read: if true;
  allow write: if false;
}

// Receipts — user can read their own, no client writes.
match /receipts/{receiptId} {
  allow read: if request.auth != null
              && resource.data.userId == request.auth.uid;
  allow write: if false;
}

// Cashback payouts — user can read their own, no client writes.
match /cashback_payouts/{payoutId} {
  allow read: if request.auth != null
              && resource.data.userId == request.auth.uid;
  allow write: if false;
}

// Review queue — never readable to clients.
match /cashback_review_queue/{queueId} {
  allow read: if false;
  allow write: if false;
}

// Merchants — public read (logo + canonical name), admin-only write.
match /merchants/{merchantId} {
  allow read: if true;
  allow write: if false;
}

// Monthly consumption aggregate — never readable to clients.
match /cashback_consumption/{month} {
  allow read: if false;
  allow write: if false;
}

// Learning labels (Sprint 0+) — never readable to clients.
match /learning/{document=**} {
  allow read: if false;
  allow write: if false;
}

// ─── ADD inside users/{userId} match block ───────────────────────────
// (paste these inside your existing `match /users/{userId} { ... }`)

// Cashback ledger — user can read own, no client writes (admin SDK only).
match /cashback_ledger/{ledgerId} {
  allow read: if request.auth != null && request.auth.uid == userId;
  allow write: if false;
}

// Purchased products — user can read own, no client writes.
match /purchased_products/{productId} {
  allow read: if request.auth != null && request.auth.uid == userId;
  allow write: if false;
}
